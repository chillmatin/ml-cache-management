\begin{frame}{Problem \\ \& Approach}
  \begin{itemize}
    \item \textbf{Challenge:} Edge caches with limited capacity must serve dynamic, bursty workloads.
    \item \textbf{Baseline policies:} LRU and LFU rely on recency/frequency heuristics; strong on Zipf workloads but blind to temporal shifts.
    \item \textbf{Our approach:} Gradient Boosting model predicts re-access probability using 8 temporal/popularity features; hybrid policy integrates ML scores with frequency.
  \end{itemize}
\end{frame}

\begin{frame}{Model \\ \& Features}
  \begin{itemize}
    \item Algorithm: Gradient Boosting (200 estimators, depth=5)
    \item Training: First 50\% of trace; test accuracy 83.3\%
    \item Features (8): recency, frequency, mean/var inter-arrival, size, normalized recency, frequency rank, time since first access
    \item Insight: Mean inter-arrival time dominates importance (55.5\%) \Rightarrow temporal patterns matter most.
  \end{itemize}
\end{frame}

\begin{frame}{Workload \\ \& Simulator}
  \begin{itemize}
    \item CDN-like traffic: Zipf \(\alpha = 1.2\), Poisson arrivals (10 req/s), flash-crowd modulation
    \item Catalog: 1{,}000 objects; Cache: 100 units; Sizes: log-normal (1--100)
    \item 5 independent runs for statistical robustness (BatchSimulator)
  \end{itemize}
\end{frame}

\begin{frame}{Results}
  \begin{columns}[t]
    \column{0.55\textwidth}
    \begin{itemize}
      \item LFU: 43.31\% \(\pm\) 5.84 hit rate (best on Zipf)
      \item LRU: 34.56\% \(\pm\) 5.30
      \item ML-Cache: 31.01\% \(\pm\) 4.35 (83.3\% prediction accuracy)
    \end{itemize}
    \column{0.45\textwidth}
    \begin{itemize}
      \item Latency: miss penalty 100 ms, hit 5 ms
      \item Observation: Frequency signal in Zipf favors LFU; ML adds temporal insight
    \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Takeaways \\ \& Next Steps}
  \begin{itemize}
    \item LFU remains strong on Zipfian workloads; ML highlights temporal drivers.
    \item Hybrid policy path: combine ML temporal scores with LFU frequency.
    \item Future: online learning, real CDN traces, ns-3 integration, distributed cache tiers.
    \item Code: \texttt{src/} (simulator, policies, models); Results: \texttt{assets/results/}
  \end{itemize}
\end{frame}
